import { Router, type IRouter } from "express";
import { db } from "@workspace/db";
import { longTermMemoryTable } from "@workspace/db/schema";
import { desc } from "drizzle-orm";
import { GetMemoryResponse } from "@workspace/api-zod";

const router: IRouter = Router();

router.get("/memory", async (_req, res) => {
  const rows = await db
    .select()
    .from(longTermMemoryTable)
    .orderBy(desc(longTermMemoryTable.strength));

  const response = GetMemoryResponse.parse({
    items: rows.map((m) => ({
      id: m.id,
      theme: m.theme,
      summary: m.summary,
      strength: m.strength,
      occurrences: m.occurrences,
      firstSeenAt: m.firstSeenAt,
      lastSeenAt: m.lastSeenAt,
    })),
  });

  res.json(response);
});

export default router;
